### Standalone function

The standalone function version of this method:

```typescript expandable lines
import { OpenRouterCore } from "@kenh-ai/sdk/core.js";
import { providersList } from "@kenh-ai/sdk/funcs/providersList.js";

// Use `OpenRouterCore` for best tree-shaking performance.
// You can create one instance of it to use across an application.
const kenhAi = new OpenRouterCore({
  httpReferer: "<value>",
  appTitle: "<value>",
  appCategories: "<value>",
  apiKey: process.env["KENH_AI_API_KEY"] ?? "",
});

async function run() {
  const res = await providersList(kenhAi);
  if (res.ok) {
    const { value: result } = res;
    console.log(result);
  } else {
    console.log("providersList failed:", res.error);
  }
}

run();
```
